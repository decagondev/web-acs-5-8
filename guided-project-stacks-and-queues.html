Valid Bracket Sequence Implementation

Step-by-Step Implementation

Step 1: Setup Bracket Maps

const bracketPairs = {
    '(': ')',
    '{': '}',
    '[': ']'
};

const openBrackets = new Set(['(', '{', '[']);
const closeBrackets = new Set([')', '}', ']']);

Step 2: Validation Function

function isValidBracketSequence(str) {
    const stack = [];
    
    for (let char of str) {
        if (openBrackets.has(char)) {
            stack.push(char);
        } else if (closeBrackets.has(char)) {
            if (stack.length === 0) return false;
            
            const lastOpen = stack.pop();
            if (bracketPairs[lastOpen] !== char) {
                return false;
            }
        }
    }
    
    return stack.length === 0;
}

Example Usage:

console.log(isValidBracketSequence("(){}[]"));  // true
console.log(isValidBracketSequence("({[]})"));  // true
console.log(isValidBracketSequence("({)}"));    // false
console.log(isValidBracketSequence("((()"));    // false
