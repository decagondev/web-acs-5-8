Implementation Steps

Step 1: Node Creation

class BSTNode {
    constructor(value) {
        this.value = value;
        this.left = null;
        this.right = null;
    }
}

Step 2: Insert Method

insert(value) {
    const newNode = new BSTNode(value);
    
    if (this.root === null) {
        this.root = newNode;
        return;
    }
    
    this._insertNode(this.root, newNode);
}

Step 3: Recursive Insert Helper

_insertNode(node, newNode) {
    if (newNode.value < node.value) {
        if (node.left === null) {
            node.left = newNode;
        } else {
            this._insertNode(node.left, newNode);
        }
    } else {
        if (node.right === null) {
            node.right = newNode;
        } else {
            this._insertNode(node.right, newNode);
        }
    }
}
